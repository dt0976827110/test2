const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

function jsonRes(p, obj) {
  const json = JSON.stringify(obj);
  if (p.callback) {
    return ContentService
      .createTextOutput(p.callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

// ===== 市場報價 =====
function getPrices(symbols) {
  const results = {};
  let usdtwd = 32;
  try {
    const fxData = JSON.parse(UrlFetchApp.fetch('https://api.exchangerate-api.com/v4/latest/USD', { muteHttpExceptions: true }).getContentText());
    usdtwd = fxData.rates?.TWD || 32;
  } catch(e) {}

  symbols.forEach(item => {
    try {
      const { symbol, type } = item;
      if (type === 'crypto') {
        if (symbol.toUpperCase() === 'USDT') {
          results[symbol] = { price: Math.round(usdtwd * 100) / 100 };
        } else {
          const pair = symbol.toUpperCase() + '-USDT';
          const url  = 'https://www.okx.com/api/v5/market/ticker?instId=' + pair;
          const res  = UrlFetchApp.fetch(url, { muteHttpExceptions: true });
          const data = JSON.parse(res.getContentText());
          const priceUsdt = parseFloat(data.data?.[0]?.last);
          results[symbol] = { price: priceUsdt ? Math.round(priceUsdt * usdtwd) : null };
        }
      } else if (type === 'gold_tw') {
        // 黃金存摺牌價估算：GC=F (USD/盎司) 換算成 NT$/公克
        // 台銀牌價 ≈ 國際金價 × 匯率 ÷ 31.1035（盎司→克）
        const url  = 'https://query1.finance.yahoo.com/v8/finance/chart/GC=F?interval=1d&range=1d';
        const data = JSON.parse(UrlFetchApp.fetch(url, { muteHttpExceptions: true }).getContentText());
        const priceUsdPerOz = data.chart?.result?.[0]?.meta?.regularMarketPrice || null;
        if (priceUsdPerOz) {
          const ntdPerGram = Math.round(priceUsdPerOz * usdtwd / 31.1035);
          results[symbol] = { price: ntdPerGram };
        } else {
          results[symbol] = { price: null };
        }
      } else if (type === 'metal_usd') {
        // 國際黃金 GC=F（USD/盎司）轉台幣
        const url = 'https://query1.finance.yahoo.com/v8/finance/chart/GC=F?interval=1d&range=1d';
        const data = JSON.parse(UrlFetchApp.fetch(url, { muteHttpExceptions: true }).getContentText());
        const priceUsd = data.chart?.result?.[0]?.meta?.regularMarketPrice || null;
        results[symbol] = { price: priceUsd ? Math.round(priceUsd * usdtwd) : null };
      } else if (type === 'metal_two') {
        const url = 'https://query1.finance.yahoo.com/v8/finance/chart/C:' + symbol + '.TWO?interval=1d&range=1d';
        const data = JSON.parse(UrlFetchApp.fetch(url, { muteHttpExceptions: true }).getContentText());
        results[symbol] = { price: data.chart?.result?.[0]?.meta?.regularMarketPrice || null };
      } else if (type === 'stock') {
        const url = 'https://query1.finance.yahoo.com/v8/finance/chart/' + symbol + '.TW?interval=1d&range=1d';
        const data = JSON.parse(UrlFetchApp.fetch(url, { muteHttpExceptions: true }).getContentText());
        results[symbol] = { price: data.chart?.result?.[0]?.meta?.regularMarketPrice || null };
      } else if (type === 'fx') {
        results[symbol] = { price: 1 };
      }
    } catch(err) {
      results[item.symbol] = { price: null, error: err.message };
    }
  });
  return results;
}

function doGet(e) {
  try {
    const p  = e.parameter;
    const ss = SpreadsheetApp.openById(SHEET_ID);

    if (p.action === 'getPrices') {
      const symbols = JSON.parse(decodeURIComponent(p.symbols || '[]'));
      const prices  = getPrices(symbols);
      return jsonRes(p, { success: true, prices });
    }

    if (p.action === 'addProduct') {
      const sheet = ss.getSheetByName('商品');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['商品名稱','種類','持有數量','現有價值','新增時間']);
      }
      sheet.appendRow(['#' + String(p.name), p.type, p.qty, p.value,
        new Date().toLocaleString('zh-TW')]);
    }

    if (p.action === 'updateProduct') {
      const sheet = ss.getSheetByName('商品');
      const data  = sheet.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        const cellName = String(data[i][0]);
        if (cellName === String(p.name) || cellName === '#' + String(p.name)) {
          sheet.getRange(i + 1, 3).setValue(p.qty);
          sheet.getRange(i + 1, 4).setValue(p.value);
          break;
        }
      }
    }

    if (p.action === 'deleteProduct') {
      const sheet = ss.getSheetByName('商品');
      const data  = sheet.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        const cellName = String(data[i][0]);
        if (cellName === String(p.name) || cellName === '#' + String(p.name)) {
          sheet.deleteRow(i + 1);
          break;
        }
      }
      return jsonRes(p, { success: true, action: 'deleted', name: p.name });
    }

    if (p.action === 'addHistory') {
      const sheet = ss.getSheetByName('歷史紀錄');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['日期時間','商品','類型','數量','價格','換算價值']);
      }
      sheet.appendRow([p.datetime, p.name, p.tradeType,
        p.qty, p.price, p.total]);
    }

    if (p.action === 'getHistory') {
      const sheet = ss.getSheetByName('歷史紀錄');
      const data  = sheet.getDataRange().getValues();
      const rows  = data.slice(1);
      return jsonRes(p, { success: true, rows });
    }

    if (p.action === 'getProducts') {
      const sheet = ss.getSheetByName('商品');
      const data  = sheet.getDataRange().getValues();
      const rows  = data.slice(1).map(r => [String(r[0]).replace(/^#/, ''), r[1], r[2], r[3], r[4]]);
      return jsonRes(p, { success: true, rows });
    }

    return jsonRes(p, { success: true });

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

確定刪除？
刪除「」後，該資產的持有數量與成本將全數清零，此操作無法復原。
原持有量 填入你目前已持有的數量。
單位依商品種類而定：
・加密貨幣：幣數（如 0.5 BTC）
・ETF/股票：股數（如 100 股）
・黃金：公克（如 3.75 克）
・現金：台幣金額（如 50000）
若是全新商品，可填 0。
購入成本 (NT$) 填入你購買此資產所花費的總金額（台幣）。

例如：
・買了 0.5 BTC，花了 NT$ 800,000，就填 800000
・買了 100 股 ETF，花了 NT$ 20,000，就填 20000

系統會用此金額計算持有均價與損益。
價格（NT$） 填入此次交易的單價（每單位的台幣價格）。

例如：
・BTC 現在 NT$ 1,600,000 一顆，就填 1600000
・股票每股 NT$ 200，就填 200

系統會自動換算總交易金額。
數量 填入此次交易的數量。
單位與商品種類相同：
・加密貨幣：幣數（如 0.1）
・ETF/股票：股數（如 50）
・黃金：公克（如 1.25）
・現金：台幣金額（如 10000）
確定解除綁定？
解除後將無法讀取或寫入 Google Sheet，重新開啟 App 後本機資料將無法還原。
確認綁定 Google Sheet？
綁定後將自動同步以下操作到你的 Google Sheet：
・新增商品 ・買入 / 賣出交易紀錄 ・刪除商品 請確認已正確部署 Apps Script 並複製了正確的網址。
